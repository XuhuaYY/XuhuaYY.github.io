集合

集合比较常见的实现方式是哈希表，这里使用的是JavaScript的Object进行封装。

集合特点

  • 无序、不可重复的元素构成
  • 数学上集合是可重复的，计算机中集合是不可重复的。
  • 集合是特殊的数组
    • 元素无顺序，无重复
    • 无顺序：不能通过下标访问；无重复：相同的元素在集合中是唯一的

封装集合

ES6中的Set是一个集合类，我们重新封装一个Set类，了解集合的底层实现。

集合常见的操作

  • has(value) 如果值在集合中，返回 true，否则返回false。
    • add(value) 向集合添加一个新的项。
  • remove(value) 从集合移除一个值。
  • clear() 移除集合中的所有项。
  • size() 返回集合所包含元素的数量。与数组的 length 属性类似。
  • values() 返回一个包含集合中所有值的数组。
  • 还有其他的方法，用的不多，这里不做封装。

封装集合类代码如下：

//使用Object类去包装Set
function Set() {
    //属性
    this.items = {}
    //has方法
    Set.prototype.has = function(value) {
        return this.items.hasOwnProperty(value);
    }
    
    //add方法
    Set.prototype.add = function(value) {
        if (this.has(value)) {
            return false;
        }
        else {
            this.items[value] = value;
            return true;
        }
    }

    //remove方法
    Set.prototype.remove = function(value)  {
        if (!this.has(value)) {
            return false;
        }
        else {
            delete this.items[value];
        }
    }

    //clear方法
    Set.prototype.clear = function() {
        this.items = {};
    }

    //size方法
    Set.prototype.size = function() {
        return Object.keys(this.items).length;
    }

    //values方法
    Set.prototype.values = function() {
        return Object.keys(this.items);
    }
}

测试代码如下:

setObj = new Set();
setObj.add(1);
setObj.add(2);
setObj.add(3);
console.log(setObj.values());	//['1', '2', '3']
console.log(setObj.size());		//3
setObj.remove(2);
console.log(setObj.values());	//['1', '3']
console.log(setObj.size());		//2
setObj.clear();
console.log(setObj.values());	//[]
console.log(setObj.size());		//0

集合间的操作

  • 并集

A集全部归为C集，B集和A集共同拥有的元素，不归于C集，否则归于C集合。最终C集合为A集和B集的并集。

  • 交集

A集和B集共同拥有的元素，归于C集，该集合为A和B集合的交集。

  • 差集

A集和B集共同拥有的元素，不归于C集，A集中的其他元素都归于C集。

  • 子集

A集全部的元素都属于B集合，则A集是B集的子集。

	/*
        集合之间的操作
    */
   // 并集 union
   Set.prototype.union = function(otherSet) {
        let unionSet = new Set();

        //SetA→unionSet
        for (let item of this.values()) {
            unionSet.items[item] = item;
        }
        //SetB(非SetA)→unionSet
        for (let item of otherSet.values()) {
            if (!this.has(item)) {
                unionSet.items[item] = item;
            }
        }
        return unionSet;
   }

   //交集 intersection
   Set.prototype.intersection = function(otherSet) {
        let interSet = new Set();

        //SetA has SetB的元素，有则添加到interSet
        for (let item of otherSet.values()) {
            if(this.has(item)) {
                interSet.items[item] = item;
            }
        }
        return interSet;
   }

   //差集 difference
   Set.prototype.difference = function(otherSet) {
        let differenceSet = new Set();

        //SetA 非B→differenceSet
        for (let item of this.values()) {
            if (!otherSet.has(item)) {
                differenceSet.items[item] = item;
            }
        }
        return differenceSet;
   } 

   //子集 subSet
   Set.prototype.subset = function(otherSet) {
        //SetA 的元素都属于 SetB,则SetA是SetB的子集
        for (let item of this.values()) {
            if (!otherSet.has(item)) {
                return false;
            }
        }
        return true;
    }

/*
    测试集合中的操作
*/
setA = new Set();
setA.add(1);
setA.add(2);
setA.add(3);
console.log(setA.values());

setB = new Set();
setB.add(3);
setB.add(4);
setB.add(5);
console.log(setB.values());

//并集
unionSet = setA.union(setB)
console.log(unionSet.values());     //['1', '2', '3', '4', '5']

//交集
interSet = setA.intersection(setB)
console.log(interSet.values());     //['3']

//差集
differenceSet = setA.difference(setB)
console.log(differenceSet.values());     //['1', '2']

//子集
setA.remove(1);
setA.remove(2);
console.log(setA.values()); 
isSubSet = setA.subset(setB);
console.log(isSubSet);  //true
